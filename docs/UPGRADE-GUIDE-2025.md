    persist(
      subscribeWithSelector((set, get) => ({
        workflows: [],
        currentWorkflow: null,

          workflows: [...state.workflows, workflow]
        })),

          workflows: state.workflows.map(w => 
            w.id === id ? { ...w, ...updates } : w
          )
        })),
        
        // Nouvelles fonctionnalit√©s
        subscribe: (selector) => {
          const store = get()
          return subscribeWithSelector(selector)(store)
        },
      })),
      { name: 'workflow-store' }
    ),
